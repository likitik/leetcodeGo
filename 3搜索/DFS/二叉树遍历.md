### 二叉树的前序遍历
```golang
/**
 * Definition for a binary tree node.
 * type TreeNode struct {
 *     Val int
 *     Left *TreeNode
 *     Right *TreeNode
 * }
 */
func preorderTraversal(root *TreeNode) []int {
    var help func(root *TreeNode)
    var res []int
    help = func(root *TreeNode){
        if root==nil{
            return
        }else{
            res = append(res,root.Val)
            help(root.Left)
            help(root.Right)
        }
    }
    help(root)
    return res
}
```
