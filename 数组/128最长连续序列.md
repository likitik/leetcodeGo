func longestConsecutive(nums []int) int {
	M := make(map[int]bool)
	for _, i := range nums {
		if _, ok := M[i]; !ok {
			M[i] = true
		}
	}
	var res int
	for _, x := range nums {
		if _, ok := M[x-1]; !ok {
			k := 0
			for {
				_, ok := M[x]
				if !ok {
					break
				}
				k++
				x++
			}
			if k > res {
				res = k
			}
		}
	}
	return res
}
